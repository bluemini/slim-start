Example of Partial Application of Functions

		<script>

		// create a function that we will partially apply
		add = function(num1, num2) {
			return num1 + num2;
		}

		// create a new function by partially applying it
		addTwo = cyn().partial(add, 2);

		// run the function and update the DOM
		cyn('res').text(addTwo(5));

		</script>
